# PRECISARAM DE AJUSTES (TENDO EM VISTA QUE A TABELA USUARIOS JÁ EXISTE), SÃO ELES:

  #PROBLEMA 01 - Ativo com defaultValue: false # Usuários existentes ficarão inativos
  # Você provavelmente quer que usuários antigos continuem válidos
  
  migrationBuilder.AddColumn<bool>(
    name: "Ativo",
    table: "Usuarios",
    type: "tinyint(1)",
    nullable: false,
    defaultValue: false); # AJUSTE PARA true

  
  #PROBLEMA 02 - DataCriacao com DateTime.MinValue # 0001-01-01 não é uma data válida para MySQL em muitos ambientes
  # Pode causar:(Erro na migration, Datas inválidas ou Comparações erradas depois)
  
  migrationBuilder.AddColumn<DateTime>(
    name: "DataCriacao",
    table: "Usuarios",
    type: "datetime(6)",
    nullable: false,
    defaultValue: new DateTime(1, 1, 1, 0, 0, 0, 0, DateTimeKind.Unspecified)); # AJUSTE PARA defaultValueSql: "CURRENT_TIMESTAMP");

  # PROBLEMA 03 - Colunas Login, Perfil, SenhaHash (nullable = false)

  # Campo NOT NULL
  # Sem defaultValue
