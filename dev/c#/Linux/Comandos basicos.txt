# Criar projeto
dotnet new console -o MeuProjeto

cd MeuProjeto
dotnet run

# Compilar projeto
dotnet build
dotnet run
# Rodar arquivo .dll
dotnet MeuProjeto.dll

# Como instalar runtime
sudo apt-get update && \
  sudo apt-get install -y aspnetcore-runtime-8.0
